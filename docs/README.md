## VueJS 

<img width="40px" src="/img/vue-logo.svg" />

``` js

<template>
    <div class="counter">
        <h1>{{number}}</h1>
        <button @click = "decrement">-</button>
        <button @click = "increment">+</button>
    </div>
</template>
<script>
    export default {
        data(){
            return{
                number: 0,
            }
        },
        methods:{
            increment(){
                if(this.number >= 0){
                    this.number++
                }
            },
            decrement(){
                if(this.number > 0){
                    this.number --
                }
            }
        }
    }
</script>

```
